#include <cstdio>
using namespace std;
typedef long long ll;
constexpr int mod=1000000007;
ll quickpow(ll x,int k,ll ret=1){
	for(; k; k>>=1,x=x*x%mod)
		(k&1)&&(ret=ret*x%mod);
	return ret;
}
constexpr int max_n=100000,max_m=100000;
constexpr int max_l=max_n+max_m+1;
int f[max_l+1];
int ifac[max_l+1],pre[max_l+1],suf[max_l+1];
int lagrange(int*const f,const int n,const int x){
	ll now=1;
	for(int i=0; i<=n; ++i)
		pre[i]=now=now*(x-i)%mod;
	now=1;
	for(int i=n; i>=0; --i)
		suf[i]=now=now*(x-i)%mod;
	now=1;
	for(int i=2; i<=n; ++i)
		now=now*i%mod;
	ifac[n]=now=quickpow(now,mod-2);
	for(int i=n; i; ifac[--i]=now)
		now=now*i%mod;
	int ans=(f[0]*ll((n&1)?mod-ifac[n]:ifac[n])%mod*suf[1]+f[n]*ll(ifac[n])%mod*pre[n-1])%mod;
	for(int i=1; i<n; ++i)
		ans=(ans+f[i]*ll(((n^i)&1)?mod-ifac[n-i]:ifac[n-i])%mod*ifac[i]%mod*pre[i-1]%mod*suf[i+1])%mod;
	return ans;
}
void test(){
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	const int l=n+m+1;
	f[0]=0;
	ll pn=0,pm=0;
	for(int i=1; i<=l; ++i){
		const ll tn=quickpow(i,n),tm=quickpow(i,m);
		f[i]=(f[i-1]+(tn-pn+mod)*(tm-pm+mod))%mod;
		pn=tn,pm=tm;
	}
	if(k<=l){
		printf("%d\n",f[k]);
		return;
	}
	printf("%d\n",lagrange(f,l,k));
}
int main(){
	int t;
	for(scanf("%d",&t); t; test(),--t);
	return 0;
}
