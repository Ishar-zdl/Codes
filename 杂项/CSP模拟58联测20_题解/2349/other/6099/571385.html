#include <bits/stdc++.h>
#define pl tr<<1
#define pr tr<<1|1
using namespace std;

const int N=1e5+10,M=1e6+10;

int n,m,q;
int a[M];
int op[M],l[M],r[M],x[M];
int va[20];
int cnt;
int cn;
bool dp[1025][N];
int v[20];
// struct nod{

map<int,int> ma;

struct node{
    int l,r,sum;
}t[M<<2];

void pushup(int tr){
    t[tr].sum=t[pl].sum|t[pr].sum;
}

void build(int l,int r,int tr){
    t[tr].l=l,t[tr].r=r;
    if(l==r) {t[tr].sum=(1<<(ma[a[l]]-1));return;}
    int mid=(l+r)>>1;
    build(l,mid,pl);
    build(mid+1,r,pr);
    pushup(tr);
}

void update(int l,int r,int k,int tr){
    if(t[tr].l==t[tr].r){
        t[tr].sum=(1<<(k-1));
        return;
    }
    int mid=(t[tr].l+t[tr].r)>>1;
    if(l<=mid) update(l,r,k,pl);
    if(mid<r) update(l,r,k,pr);
    pushup(tr);
}

int query(int l,int r,int tr){
    if(l<=t[tr].l&&t[tr].r<=r){
        return t[tr].sum;
    }
    int res=0;
    int mid=(t[tr].l+t[tr].r)>>1;
    if(l<=mid) res=query(l,r,pl);
    if(mid<r) res|=query(l,r,pr);
    return res;
}

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0),cout.tie(0);
    cin>>n>>m>>q;
    for(int i=1;i<=n;i++) {
        cin>>a[i];
        if(!(ma[a[i]])) ma[a[i]]=++cnt,va[cnt]=a[i];
    }
    for(int i=1;i<=q;i++){
        cin>>op[i]>>l[i]>>r[i]>>x[i];
        if(op[i]==1&&!ma[x[i]]) ma[x[i]]=++cnt,va[cnt]=a[i];
    }
    for(int i=0;i<(1<<cnt);i++){
        cn=0;
        for(int j=0;j<cnt;j++){
            if((i>>j)&1){
                v[++cn]=va[j+1];
            }
        }
        dp[i][0]=1;
        for(int j=1;j<=cn;j++){
            for(int k=v[j];k<=m;k++){
                dp[i][k]|=dp[i][k-v[j]];
            }
        }
    }
    build(1,n,1);
    for(int i=1;i<=q;i++){
        if(op[i]==1){
            update(l[i],r[i],ma[x[i]],1);
        }
        else{
            cout<<(dp[query(l[i],r[i],1)][x[i]]?"Yes\n":"No\n");
        }
    }
    
}
