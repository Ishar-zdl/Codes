#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 1001000;

int n, m, q;
int a[N], b[N], top, c[N], d[N], tp;
bool ji, ou;
bool flag, has_answer;

int find(int x, int n){
	int l = 0, r = n-1;
	while(l < r){
		int mid = l+r>>1;
		if(d[mid] > x) r = mid;
		else l = mid + 1;
	}
	
	return r+1;
}

bool check2(int tp, int x){
//	cout<<1<<endl;
//	int count = 0;
	
	for(int i = tp-1; i >= 0; i --){
//		if(count ++ >= 10) return true;
		if(x == d[i]) return true;
		else if(x > d[i]){
			x -= d[i];
			int r = find(x, tp);
			if(check2(r, x)) return true;
			x += d[i];
		}
		
	}
	
	return false;
}

bool check1(int top, int x){

	sort(c, c+top);


	for(int i = 0; i < top; i ++){
		if(c[i] == 2){
			if(ji) return true; 
		}
	}
		
//	int count = 0;

	for(int i = 0; i < top; i ++){
		
		int j = c[i], k = c[i];
		while(k < x){
			d[tp++] = k; 
			k += j;
		}
		if(k == x) return true;
 	}
 	
	sort(d, d+tp);
	
	if(check2(tp, x)) return true;
	
	
	return false;
}

void modify(int l, int r, int x){
	if(l == r){
		a[r] = x;
		return;	
	}
	int mid = l+r>>1;
	modify(l, mid, x), modify(mid+1, r, x);
}

void query(int l, int r, int x){
	if(l == r){
		if(a[r] > x) return;
		if(b[a[r]]) return;
		if(x % a[r] == 0){
			flag = true;
			return;
		}
		if(a[r] % 2) ji = 1;
		b[a[r]] = 1;
		c[top ++] = a[r]; 
		has_answer = true;
		return;
	}
	int mid = l+r>>1;
	query(l, mid, x), query(mid+1, r, x);
	
}

int main(){
	
	//freopen("t1.in", "r", stdin);
	//freopen("t1.out", "w", stdout);
	
	scanf("%d%d%d", &n, &m, &q);
	
	for(int i = 1; i <= n; i ++) scanf("%d", &a[i]);
	

	while(q --){
		int t, l, r, x;
		scanf("%d%d%d%d", &t, &l, &r, &x);
		if(t == 1){
			modify(l, r, x);
//			for(int i = l; i <= r; i ++) cout<<a[i]<<" ";
//			cout<<"\n";
		}
		else{
			int z = 0;
			flag = false, has_answer = false;
			top = 0;
			query(l, r, x);
			if(flag){
				printf("Yes\n");
				for(int i = 0; i < top; i ++) b[c[i]] = 0; 
			}
			else{
				if(has_answer){
					if(check1(top,x)) printf("Yes\n");	
					else printf("No\n");
				}
				else printf("No\n");
			}
		}
	}
	
	
	return 0;
}
