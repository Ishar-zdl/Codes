const int o=5e5+10;
int n,m,q,U[o*2],V[o*2],W[o*2],h[o],cnt,d[o],anc[o][20],stu[o][20],stk[o][20],tp[o];bool vis[o];
struct Edge{int v,p;}e[o*4];
inline void ad(int U,int V){e[++cnt].v=V;e[cnt].p=h[U];h[U]=cnt;}
inline void ins(int*arr,int v){for(int i=0;i<3&&arr[i]!=v;++i) if(!arr[i]){arr[i]=v;break;}}
struct info{int bg[3],ans[3],ed[3][3];int calc(){return max(ans[0],max(ans[1],ans[2]));}}z[o][20],Ans[o];
info merge(info A,info B){
	if(A.bg[0]<0) return B;
	info res;
	for(int i=0;i<3;++i) res.ans[i]=0;
	for(int i=0;i<3;++i) if(res.bg[i]=A.bg[i]){
		res.ans[i]=0;
		for(int j=0;j<3;++j) res.ed[i][j]=0;
		for(int j=0;j<3;++j) if(A.ed[i][j]) for(int k=0,t;k<3;++k) if(B.bg[k]){
			if((t=A.ans[i]+B.ans[k]+(A.ed[i][j]!=B.bg[k]))>res.ans[i]){res.ans[i]=t;for(int j=0;j<3;++j) res.ed[i][j]=0;}
			if(t==res.ans[i]) for(int $=0;$<3&&B.ed[k][$];++$) ins(res.ed[i],B.ed[k][$]);
		}
	}
	return res;
}
void dfs(int nw){
	d[nw]=d[anc[nw][0]]+1;vis[nw]=1;
	for(int i=h[nw];i;i=e[i].p) if(!vis[e[i].v]) anc[e[i].v][0]=nw,dfs(e[i].v);
}
int main(){

	for(int i=1;i<=m;++i) scanf("%d%d%d",&U[i],&V[i],&W[i]),ad(U[i],V[i]),ad(V[i],U[i]);

		for(int j=0;j<3;++j) if(z[U[i]][0].bg[j]==W[i]) ins(z[U[i]][0].ed[j],W[i]);

	scanf("%d",&q);
