#include<iostream>
#include<cstdio>
#include<vector>
#include<set>
#include<map>
#include<cstring>
using namespace std;
int n,m,q,cnt,tot,h[500001],dep[500001],fa[500001][19];
struct edge
{
    int nxt,to;
}e[500001<<1];
struct node
{
    struct element
    {
        int l,r,ans;
    };
    element a[10];
    int cnt;
    inline bool empty()
    {
        return !cnt;
    }
    inline void clear()
    {
        memset(a,0,sizeof a);
        cnt=0;
    }
}val[500001][19],st[500001];
map<pair<int,int>,int> id;
inline int read()
{
    int x=0;
    char c=getchar();
    while(c<'0'||c>'9')
        c=getchar();
    while(c>='0'&&c<='9')
    {
        x=(x<<1)+(x<<3)+(c^48);
        c=getchar();
    }
    return x;
}
inline void add(int x,int y)
{
    e[++tot]={h[x],y};
    h[x]=tot;
}
inline void merge(node &x,node &y,node &res)
{
    if(x.empty())
    {
        res=y;
        return;
    }
    if(y.empty())
    {
        res=x;
        return;
    }
    map<pair<int,int>,int> vis;
    res.clear();
    int a=x.cnt,b=y.cnt;
    for(int i=0;i<a;++i)
        for(int j=0;j<b;++j)
            if(!vis.count({x.a[i].l,y.a[j].r}))
                res.a[vis[{x.a[i].l,y.a[j].r}]=res.cnt++]={x.a[i].l,y.a[j].r,x.a[i].ans+y.a[j].ans+(x.a[i].r!=y.a[j].l)};
            else
            {
                int tmp=vis[{x.a[i].l,y.a[j].r}];
                res.a[tmp].ans=max(res.a[tmp].ans,x.a[i].ans+y.a[j].ans+(x.a[i].r!=y.a[j].l));
            }
}
inline void dfs(int k,int f,int deep)
{
    dep[k]=deep;
    fa[k][0]=f;
    if(k<f)
        val[k][0]=st[id[{k,f}]];
    else
        val[k][0]=st[id[{f,k}]];
    for(int i=1;i<=18;++i)
    {
        fa[k][i]=fa[fa[k][i-1]][i-1];
        merge(val[k][i-1],val[fa[k][i-1]][i-1],val[k][i]);
    }
    for(int i=h[k];i;i=e[i].nxt)
    {
        if(e[i].to==f)
            continue;
        dfs(e[i].to,k,deep+1);
    }
}
inline int LCA(int x,int y)
{
    if(dep[x]<dep[y])
        swap(x,y);
    for(int d=dep[x]-dep[y],i=0;1<<i<=d;++i)
        if(d>>i&1)
            x=fa[x][i];
    if(x==y)
        return x;
    for(int i=18;~i;--i)
        if(fa[x][i]^fa[y][i])
        {
            x=fa[x][i];
            y=fa[y][i];
        }
    return fa[x][0];
}
inline int query(int x,int y)
{
    int lca=LCA(x,y);
    node res1,res2,tmp;
    res1.clear();
    res2.clear();
    tmp.clear();
    for(int i=18;~i;--i)
        if(dep[fa[x][i]]>=dep[lca])
        {
            merge(res1,val[x][i],tmp);
            res1=tmp;
            x=fa[x][i];
        }
    for(int i=18;~i;--i)
        if(dep[fa[y][i]]>=dep[lca])
        {
            merge(res2,val[y][i],tmp);
            res2=tmp;
            y=fa[y][i];
        }
    for(int i=0;i<res2.cnt;++i)
        swap(res2.a[i].l,res2.a[i].r);
    merge(res1,res2,tmp);
    int res=0;
    for(int i=0;i<tmp.cnt;++i)
        res=max(res,tmp.a[i].ans);
    return res;
}
int main()
{
    n=read(),m=read();
    for(int i=1;i<=m;++i)
    {
        int x=read(),y=read(),w=read();
        if(x>y)
            swap(x,y);
        if(!id.count({x,y}))
        {
            st[id[{x,y}]=++cnt].clear();
            add(x,y);
            add(y,x);
        }
        int p=id[{x,y}];
        if(st[p].cnt>=3)
            continue;
        bool flag=1;
        for(int i=0;i<st[p].cnt;++i)
            if(st[p].a[i].l==w)
            {
                flag=0;
                break;
            }
        if(flag)
            st[p].a[st[p].cnt++]={w,w,0};
    }
    dfs(1,0,1);
    q=read();
    while(q--)
    {
        int x=read(),y=read();
        cout<<query(x,y)<<'\n';
    }
    return 0;
}
