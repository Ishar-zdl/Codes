using namespace std;
const int N=5e5+5,M=1e6+5;
vector<int>v[N],tid[N];
vector<int>col[N];
int cnt;
struct node
{
	int u,v,w;
}b[M];
bool cmp(node x,node y)
{
	if(x.u!=y.u)return x.u<y.u;
	if(x.v!=y.v)return x.v<y.v;
	return x.w<y.w;
}
int dk[N],yh[N];
int fa[N][20];
int tf[N],tb[N];
int mdep[N],cg[N];
int dep[N];
void dfs(int x,int f)
{
	tf[x]=f;
	dep[x]=dep[f]+1;
	fa[x][0]=f;
	for(int i=1;i<=19;i++)fa[x][i]=fa[fa[x][i-1]][i-1];
	for(int i=0;i<v[x].size();i++)
	{
		if(v[x][i]==f)continue;
		tb[v[x][i]]=tid[x][i];
		yh[v[x][i]]=yh[x];
		int sth=tid[x][i];
		for(int j=0;j<col[sth].size();j++)yh[v[x][i]]^=col[sth][j];
		dfs(v[x][i],x);
	}
}
int lca(int x,int y)
{
	if(dep[x]<dep[y])swap(x,y);
	for(int i=19;i>=0;i--)
	{
		if(dep[fa[x][i]]>=dep[y])x=fa[x][i];
	}
	if(x==y)return x;
	for(int i=19;i>=0;i--)
	{
		if(fa[x][i]!=fa[y][i])
		{
			x=fa[x][i];
			y=fa[y][i];
		}
	}
	return fa[x][0];
}
int zh[500005],cgh[500005];
void dfs2(int x,int f)
{
	zh[x]=zh[f];
	cgh[x]=cgh[f];
	if(col[tb[x]].size()==1)zh[x]++;
	if(cg[x]==1)cgh[x]++;
	for(int i=0;i<v[x].size();i++){
		if(v[x][i]==f)continue;
		dfs2(v[x][i],x);
	}
}
int findlas(int x,int f)
{
	if(zh[x]-zh[f]==0)return -1;
	for(int i=19;i>=0;i--)
	{
		if(dep[fa[x][i]]<dep[f])continue;
		if(zh[fa[x][i]]-zh[f]==0)continue;
		x=fa[x][i];
	}
	return x;
}
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=m;i++)
	scanf("%d%d%d",&b[i].u,&b[i].v,&b[i].w);
	sort(b+1,b+m+1,cmp);
	for(int i=1;i<=m;i++)
	{
		if(i==1||b[i].u!=b[i-1].u||b[i].v!=b[i-1].v)
		{
			v[b[i].u].push_back(b[i].v);
			v[b[i].v].push_back(b[i].u);
			++cnt;
			tid[b[i].u].push_back(cnt);
			tid[b[i].v].push_back(cnt);
			col[cnt].push_back(b[i].w);
		}
		else if(b[i].w!=b[i-1].w)
		{
			col[cnt].push_back(b[i].w);
		}
	}
	dfs(1,0);
	for(int i=1;i<=n;i++)
	{
		if(col[tb[i]].size()==1)
		{
			int now=col[tb[i]][0],sth=tf[i];
			while(sth!=1)
			{
				if(col[tb[sth]].size()==1)
				{
					mdep[i]=dep[tf[sth]];
					if(now==col[tb[sth]][0])
					{
						cg[i]=1;
					}
					break;
				}
				if(col[tb[sth]].size()>2)
				{
					mdep[i]=dep[sth];
					cg[i]=0;
					break;
				}
				if(col[tb[sth]].size()==2)
				{
					if(col[tb[sth]][0]!=now&&col[tb[sth]][1]!=now)
					{
						mdep[i]=dep[sth];
						cg[i]=0;
						break;
					}
					now=col[tb[sth]][0]+col[tb[sth]][1]-now;
				}
				sth=tf[sth];
			}
			if(sth==1)
			{
				mdep[i]=dep[1];
				cg[i]=0;
			}
		}
	}
	dfs2(1,0);
	int q,x,y,lc;
	scanf("%d",&q);

	{
		scanf("%d%d",&x,&y);
		if(x==y)
		{
			printf("0\n");
			continue;

		lc=lca(x,y);
		int ans=dep[x]+dep[y]-2*dep[lc]-1;
		if(ans==0)
		{
			printf("0\n");
			continue;
		}
		int tx=findlas(x,lc),ty=findlas(y,lc);
		if(tx!=-1)
		{
			ans-=(cgh[x]-cgh[tx]);
		}
		if(ty!=-1)
		{
			ans-=(cgh[y]-cgh[ty]);
		}
		if(tx==-1||ty==-1)
		{
			printf("%d\n",ans);
			continue;
		}
		if(dep[lc]<mdep[tx]||dep[lc]<mdep[ty])
		{
			printf("%d\n",ans);
			continue;
		}
		if((yh[tx]^yh[lc])==(yh[ty]^yh[lc]))ans--;
		printf("%d\n",ans);
	}
	return 0;
}
