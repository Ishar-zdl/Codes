inline int read()

}

inline void eadd(int x,int y,int w)
{
	e[++etot].to=y;
	e[etot].nxt=h[x];

void dfs(int x)
{
	dfn[x]=++dcnt;
	ga[x][0]=x;
	for(int b=0;b<t[x].cnt;b++)f[x][0][b][b]=0;
	for(int j=1;(1<<j)<=dep[x];j++)
	{
		fa[x][j]=fa[fa[x][j-1]][j-1];
		ga[x][j]=fa[ga[x][j-1]][j-1];
		int y=ga[x][j],u=fa[x][j-1],v=ga[x][j-1];
		for(int a=0;a<t[x].cnt;a++)
		for(int b=0;b<t[y].cnt;b++)
		for(int p=0;p<t[v].cnt;p++)
		for(int q=0;q<t[u].cnt;q++)
		upd(f[x][j][a][b],f[x][j-1][a][p]+f[u][j-1][q][b]+(t[u].d[q]!=t[v].d[p]));

	}
}
